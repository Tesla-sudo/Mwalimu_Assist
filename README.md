# Eco_Collect_Rewards1

# 1. Backend Setup
Installation of dependencies 
cd Backend

npm install express mongoose dotenv cors openai morgan jsonwebtoken joi helmet axios compression bcryptjs @sendgrid/mail @google/generative-ai @google/genai express-rate-limit debug socket.io http 

npm install --save-dev nodemon

npm run dev


# 2. Frontend setup
cd client

npm install axios react react-dom react-router-dom tailwindcss socket.io-client recharts lucide-react html2canvas jspdf jspdf-autotable

npm run dev
